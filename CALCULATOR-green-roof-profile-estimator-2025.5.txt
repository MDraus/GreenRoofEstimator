        function showModal(title, message, isConfirm) {
            document.getElementById('modal-title').textContent = title;
            document.getElementById('modal-message').textContent = message;

        function loadData() {
            greenRoofData = JSON.parse(localStorage.getItem('greenRoofData') || JSON.stringify(defaultGreenRoofData));

        }

        async function saveData() {
            var oldData = JSON.parse(localStorage.getItem('greenRoofData')) || defaultGreenRoofData;
            var newData = Array.from(App.dataTableBody.querySelectorAll('tr')).map(function(row) {
                var itemData = getItemById(row.dataset.id);
                return {
                    id: row.dataset.id,
                    category: row.querySelector('[data-field="category"]').value,
                    item: row.querySelector('[data-field="item"]').value,
                    base_cost: parseFloat(row.querySelector('[data-field="base_cost"]').value) || 0,
                    hoisting_sf_per_pallet: parseFloat(row.querySelector('[data-field="hoisting_sf_per_pallet"]').value) || 0,
                    install_rate_sf_per_mh: parseFloat(row.querySelector('[data-field="install_rate_sf_per_mh"]').value) || 0,
                    depth_inches: parseFloat(row.querySelector('[data-field="depth_inches"]').value) || 0,
                    dry_weight_psf: parseFloat(row.querySelector('[data-field="dry_weight_psf"]').value) || 0,
                    saturated_weight_psf: parseFloat(row.querySelector('[data-field="saturated_weight_psf"]').value) || 0,
                    storage_type: row.querySelector('[data-field="storage_type"]').value,
                    storage_ratio: parseFloat(row.querySelector('[data-field="storage_ratio"]').value) || 0,
                    cost_unit: itemData ? itemData.cost_unit : 'sf'
                };
            });
            logChanges(oldData, newData);
            greenRoofData = newData;
            localStorage.setItem('greenRoofData', JSON.stringify(greenRoofData));
            
            globalDefaults.laborRate = parseFloat(document.getElementById('defaultLaborRate').value) || 0;
            globalDefaults.markup1 = parseFloat(document.getElementById('defaultMarkup1').value) || 0;
            globalDefaults.markup2 = parseFloat(document.getElementById('defaultMarkup2').value) || 0;
            globalDefaults.craneCost = parseFloat(document.getElementById('defaultCraneCost').value) || 0;
            globalDefaults.crewSize = parseFloat(document.getElementById('defaultCrewSize').value) || 0;
            globalDefaults.palletsPerDay = parseFloat(document.getElementById('defaultPalletsPerDay').value) || 0;
            localStorage.setItem('greenRoofGlobalDefaults', JSON.stringify(globalDefaults));
            
            await showModal('Success', 'Data and default settings have been saved successfully.');
            renderAll();
        }

        function logChanges(oldData, newData) {
            var changes = [];
            newData.forEach(function(newItem) {
                var oldItem = oldData.find(function(item) { return item.id === newItem.id; });
                if (!oldItem) {
                    changes.push({ item: newItem.item, field: 'newItem', oldValue: null, newValue: 'Item Added' });
                } else {
                    Object.keys(newItem).forEach(function(key) {
                        if (key !== 'id' && String(newItem[key]) !== String(oldItem[key])) {
                            changes.push({ item: newItem.item, field: key, oldValue: oldItem[key], newValue: newItem[key] });
                        }
                    });
                }
            });
            if (changes.length > 0) {
                changeHistory.unshift({ timestamp: new Date().toISOString(), changes: changes });
                localStorage.setItem('changeHistory', JSON.stringify(changeHistory));
                renderChangeHistory();
            }
        }

                localStorage.removeItem('changeHistory');
                localStorage.removeItem('greenRoofGlobalDefaults');
                location.reload();
            }
        }

        }

        function handleProfileChange(profileId) {
            var form = document.getElementById('profile-' + profileId);
            if (!form) return;

            // Update Total Depth
            var totalDepth = 0;
            form.querySelectorAll('select.component-input, input.component-input:checked').forEach(function(input) {
                var item = getItemById(input.value);
                if (item && item.category !== 'drain enclosures' && item.category !== 'edging') {
                    totalDepth += item.depth_inches;
                }
            });
            document.getElementById('total-depth-display-' + profileId).textContent = totalDepth.toFixed(2);
            
            // Auto-select drain enclosure item based on edging
            var edgingSelect = form.querySelector('#edging-select-' + profileId);
            var selectedEdgingOption = edgingSelect.options[edgingSelect.selectedIndex];
            var edgingDepth = parseFloat(selectedEdgingOption.dataset.depth) || 0;
            var drainType = form.querySelector('input[name="drain_enclosure_type-' + profileId + '"]:checked').value;
            
            // Uncheck all hidden drain radios
            form.querySelectorAll('input[name="drain_enclosure_item-' + profileId + '"]').forEach(function(r) { r.checked = false; });

            if (edgingDepth > 0 && drainType !== 'none') {
                 var drainItem = greenRoofData.find(function(i) { return i.category === 'drain enclosures' && i.type === drainType && i.depth_inches === edgingDepth; });
                 if (drainItem) {
                    var drainRadio = form.querySelector('input[value="' + drainItem.id + '"]');
                    if(drainRadio) drainRadio.checked = true;
                 }
            }

            updateMultiplierNotes(profileId);
        }

        async function calculateAllCosts() {
            var profiles = App.profilesContainer.querySelectorAll('.profile-form');
            if (profiles.length === 0) {
                await showModal('No Profiles', 'Please add at least one profile to calculate.');
                return;
            }
            
            lastCalculationResults = Array.from(profiles).map(function(form) {
                var profileId = form.dataset.profileId;
                var roofArea = parseFloat(form.querySelector('.profile-roof-area').value) || 0;
                var breakdown = [];
                var totals = { material: 0, hoisting: 0, install: 0, fi: 0 };
                var performance = { dry_weight_psf: 0, saturated_weight_psf: 0, retention_capacity_in: 0, detention_capacity_in: 0, retention_capacity_cf: 0, detention_capacity_cf: 0 };
                var aggregated = {};
                var g = Object.assign({}, globalDefaults);
                var costPerPallet = safeDivide(g.craneCost, g.palletsPerDay) + safeDivide(g.crewSize * 8 * g.laborRate, g.palletsPerDay);

                form.querySelectorAll('select.component-input, input.component-input:checked').forEach(function(input) {
                    var item = getItemById(input.value);
                    if (!item) return;

                    var materialCostSf = 0;
                    var quantity = 0;
                    var multiplier = 1;

                    if(item.category === 'edging') {
                        multiplier = parseFloat(form.querySelector('.edging_multiplier').value) || 0;
                        materialCostSf = item.base_cost * multiplier;
                        quantity = roofArea * multiplier;
                    } else if (item.category === 'drain enclosures') {
                         multiplier = parseFloat(form.querySelector('.drain_multiplier').value) || 0;
                         quantity = roofArea * multiplier;
                         materialCostSf = safeDivide(item.base_cost * quantity, roofArea);
                    } else {
                        materialCostSf = item.base_cost;
                        quantity = roofArea;
                    }
                    
                    var hoistingCostSf = safeDivide(costPerPallet, item.hoisting_sf_per_pallet);
                    var installCostSf = safeDivide(g.laborRate, item.install_rate_sf_per_mh);
                    var totalInstalledCostSf = materialCostSf + hoistingCostSf + installCostSf;
                    var fiPrice = totalInstalledCostSf * (1 + safeDivide(g.markup1, 100)) * (1 + safeDivide(g.markup2, 100));
                    
                    breakdown.push({ item: item, quantity: quantity, materialCostSf: materialCostSf, hoistingCostSf: hoistingCostSf, installCostSf: installCostSf, furnishedInstalledPrice: fiPrice });
                    totals.material += materialCostSf;
                    totals.hoisting += hoistingCostSf;
                    totals.install += installCostSf;
                    totals.fi += fiPrice;
                    
                    if (item.category !== 'edging' && item.category !== 'drain enclosures') {
                        performance.dry_weight_psf += item.dry_weight_psf || 0;
                        performance.saturated_weight_psf += item.saturated_weight_psf || 0;
                        if (item.storage_type === 'retention') {
                            performance.retention_capacity_in += (item.depth_inches * (item.storage_ratio || 0));
                        } else if (item.storage_type === 'detention') {
                            performance.detention_capacity_in += (item.depth_inches * (item.storage_ratio || 0));
                        }
                    }

                    if (!aggregated[item.category]) {
                        aggregated[item.category] = { cost: 0, descriptions: [] };
                    }
                    aggregated[item.category].cost += fiPrice;
                    aggregated[item.category].descriptions.push(item.item);
                });

                performance.retention_capacity_cf = (performance.retention_capacity_in / 12) * roofArea;
                performance.detention_capacity_cf = (performance.detention_capacity_in / 12) * roofArea;

                Object.keys(aggregated).forEach(function(cat) {
                    aggregated[cat].description = aggregated[cat].descriptions.join(', ');
                });

                return { name: form.querySelector('.profile-name').value || 'Profile ' + profileId, roofArea: roofArea, breakdown: breakdown, totals: totals, performance: performance, aggregated: aggregated, totalCostPerSf: totals.fi, totalProjectCost: totals.fi * roofArea, globalsUsed: g };
            });
            
            renderDetailedResults(lastCalculationResults);
            renderEstimatingView(lastCalculationResults);
            var estimatingTab = document.getElementById('estimating-tab');
            if(lastCalculationResults.length > 0) {
                 estimatingTab.classList.remove('tab-disabled');
            } else {
                 estimatingTab.classList.add('tab-disabled');
            }
        }

        function exportToCSV() {
            if (lastCalculationResults.length === 0) {
                showModal('No Results', 'No results to export. Please calculate first.');
                return;
            }

            var sanitizeCsvField = function(field) {
                var sanitized = String(field);
                if (sanitized.search(/("|,|\n)/g) >= 0) {
                    sanitized = '"' + sanitized.replace(/"/g, '""') + '"';
                }
                return sanitized;
            };

            var csvContent = "data:text/csv;charset=utf-8,";
            var headers = "Profile Name,Item Category,Item Name,Quantity,Unit,Material Cost/SF,Hoisting Cost/SF,Install Cost/SF,F&I Cost/SF\n";
            csvContent += headers;
            lastCalculationResults.forEach(function(profile) {
                profile.breakdown.forEach(function(lineItem) {
                    var row = [
                        sanitizeCsvField(profile.name),
                        sanitizeCsvField(lineItem.item.category),
                        sanitizeCsvField(lineItem.item.item),
                        lineItem.quantity.toFixed(2),
                        sanitizeCsvField(lineItem.item.cost_unit.toUpperCase()),
                        lineItem.materialCostSf.toFixed(2),
                        lineItem.hoistingCostSf.toFixed(2),
                        lineItem.installCostSf.toFixed(2),
                        lineItem.furnishedInstalledPrice.toFixed(2)
                    ].join(',');
                    csvContent += row + "\n";
                });
            });

            var encodedUri = encodeURI(csvContent);
            var link = document.createElement("a");
            link.setAttribute("href", encodedUri);
            link.setAttribute("download", 'green_roof_estimate_' + new Date().toISOString().split('T')[0] + '.csv');
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }
